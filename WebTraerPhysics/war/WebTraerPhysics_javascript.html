		function sketchProc(processing) {
		
		var physics;

		var p;
		var anchor;
		var s;		
		  
			processing.setup = function() {
				processing.size( 400, 400 );
				processing.smooth();
				processing.fill( 0 );
				processing.ellipseMode( processing.CENTER );

				physics =  new TraerPhysics.ParticleSystem( 1, 0.05 );
				
				p = physics.makeParticle( 1.0, processing.width/2, processing.height/2, 0 );
				anchor = physics.makeParticle( 1.0, processing.width/2, processing.height/2, 0 );
				anchor.makeFixed(); 
				s = physics.makeSpring( p, anchor, 0.5, 0.1, 75 );

			};
			
			processing.mousePressed = function() {
				p.makeFixed(); 
				p.position().set( processing.mouseX, processing.mouseY, 0 );
			};

			processing.mouseDragged = function() {
				p.position().set( processing.mouseX, processing.mouseY, 0 );
			};

			processing.mouseReleased = function() {
			   p.makeFree(); 
			};			

			processing.draw = function() {

				physics.tick();

				processing.background( 255 );

				processing.line( p.position().x(), p.position().y(), anchor.position().x(), anchor.position().y() );
				processing.ellipse( anchor.position().x(), anchor.position().y(), 5, 5 );
				processing.ellipse( p.position().x(), p.position().y(), 20, 20 );
			};
	  
		  
		}

function sketchProc(processing) {



	var w = 800;
	var h = 500;
	
	var agent;
	var terseHandler;
	var circles;	
	
	var eventHandler;

	processing.setup = function() {
	
		
		
		processing.size(w,h);			

		terseHandler = new TerseHandler();	
		
		
		agent = new MouseAgent(terseHandler, "my_mouse");
		eventHandler = new JsEventHandler(processing.externals.canvas, agent, "mouseEvent");
		
		
		circles = new Array(50);
		for (var i = 0; i < circles.length; i++)
			circles[i] = new GrabbableCircle(agent,this);
		
		
	};

	processing.draw = function() {
	
	  

		  processing.background(0);
		  for (var i = 0; i < circles.length; i++){
			if ( circles[i].grabsAgent(agent) )
			  circles[i].draw(processing.color(255, 0, 0));
			else
			  circles[i].draw();
		  }
		  terseHandler.handle();
		  

		  
	};
  
}
